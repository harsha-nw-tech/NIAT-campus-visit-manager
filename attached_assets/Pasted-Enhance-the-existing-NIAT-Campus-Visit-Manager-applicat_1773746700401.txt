# 🧠 STEP 2: CREATE TEMPLATE SERVICE

Create:

```js
server/services/templateService.js
```

```js
import { TEMPLATE_CONFIG } from "../config/templateConfig.js";
import { getConfig } from "./niatClient.js";

export async function updateTemplate(applicationId) {
  const { baseUrl, clientKeyDetailsId } = getConfig();
  const { TEMPLATE_ID, SECTION_ID, FIELDS } = TEMPLATE_CONFIG;

  const url = `${baseUrl}/api/nw_application/application/template_response/update/v1/`;

  for (const field of FIELDS) {
    const payload = {
      application_id: applicationId,
      template_id: TEMPLATE_ID,
      section_id: SECTION_ID,
      field_id: field.FIELD_ID,
      field_value: field.FIELD_VALUE
    };

    const response = await fetch(url, {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        "x-api-key": process.env.GAMMA_NIAT_API_KEY
      },
      body: JSON.stringify({
        clientKeyDetailsId: clientKeyDetailsId,
        data: `'${JSON.stringify(payload)}'`
      })
    });

    if (!response.ok) {
      const errorText = await response.text();
      throw new Error(`Template update failed: ${errorText}`);
    }
  }
}
```
